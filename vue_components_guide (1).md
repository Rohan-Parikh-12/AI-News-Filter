## 3. ApiService Setup

```typescript
// The store uses ApiService from @/@core/services/ApiService
// NOT axios directly — ApiService.setHeader() injects the token

// resources/ts/@core/services/ApiService.ts (already in template)
import axios from 'axios'

const ApiService = {
  setHeader() {
    const token = localStorage.getItem('token')
    axios.defaults.headers.common['Authorization'] = `Bearer ${token}`
    axios.defaults.headers.common['Accept'] = 'application/json'
  },
  get(url: string, data?: any) {
    this.setHeader()
    return axios.get(url, data)
  },
  post(url: string, data?: any) {
    this.setHeader()
    return axios.post(url, data)
  },
}

export default ApiService
```

## 4. Pinia Store — TypeScript

```typescript
// resources/ts/stores/category.ts
// Generated by: generate-store category
// Pattern from: resources/ts/stores/employeeMaster.ts

import { ref } from 'vue'
import { defineStore } from 'pinia'
import ApiService from '@/@core/services/ApiService'
import type { CategoryListParams } from '@/types/category'

export const useCategoryStore = defineStore('category', () => {

  const errors   = ref<string[]>([])
  const message  = ref<string | null>(null)
  const isLoading = ref(false)

  // ── Error handler ──────────────────────────────────────────────────────
  const handleError = (error: any) => {
    errors.value = error.response?.data?.errors || 'An unexpected error occurred.'
  }

  // ── Base API wrapper — used by ALL store methods ───────────────────────
  const apiCall = async (method: 'get' | 'post', url: string, data?: any) => {
    ApiService.setHeader()
    try {
      const response = await ApiService[method](url, data)
      return response.data
    } catch (error) {
      handleError(error)
      return { success: false, errors }
    }
  }

  // ── 1. Statistics — for 4 stat cards on listing page ──────────────────
  const categoryStatistic = async () => {
    return await apiCall('get', '/api/categoryStatistic')
  }

  // ── 2. Get paginated data — for VDataTableServer ───────────────────────
  const getCategoryData = async (data: CategoryListParams) => {
    const queryString = new URLSearchParams(data as any).toString()
    return await apiCall('get', `/api/getCategoryData?${queryString}`)
  }

  // ── 3. Get single record — for edit drawer ─────────────────────────────
  const getCategoryDetails = async (id: number) => {
    const queryString = new URLSearchParams({ id: String(id) }).toString()
    return await apiCall('get', `/api/getCategoryDetails?${queryString}`)
  }

  // ── 4. Create OR Update — id present = update, no id = create ─────────
  const manageCategory = async (value: any) => {
    return await apiCall('post', '/api/manageCategory', value)
  }

  // ── 5. Delete — single { id } or bulk { ids: [] } ─────────────────────
  const categoryDelete = async (value: { id?: number; ids?: number[] }) => {
    return await apiCall('post', '/api/categoryDelete', value)
  }

  // ── 6. Status change — { id, newStatus: '0'|'1'|'2' } ────────────────
  const categoryStatusChange = async (value: any) => {
    return await apiCall('post', '/api/categoryStatusChange', value)
  }

  // ── 7. Active list — for dropdowns in other module drawers ────────────
  const getActiveCategoryList = async (data: any = {}) => {
    const queryString = new URLSearchParams(data).toString()
    return await apiCall('get', `/api/getActiveCategoryList?${queryString}`)
  }

  return {
    errors,
    message,
    isLoading,
    categoryStatistic,
    getCategoryData,
    getCategoryDetails,
    manageCategory,
    categoryDelete,
    categoryStatusChange,
    getActiveCategoryList,
  }
})
```

## 6. Listing Table Component

```vue
<!-- resources/ts/views/categories/CategoryListingTable.vue -->
<!-- Generated by: generate-listing-table Category -->
<!-- Pattern from: views/settings/employee-departments/DepartmentListingTable.vue -->

<script setup lang="ts">
import { computed, nextTick, onMounted, ref, watch } from 'vue'
import Swal from 'sweetalert2'
import { useCategoryStore } from '@/stores/category'
import { useAuthStore }     from '@/stores/auth'
import { useSnackbar }      from '@/composables/useSnackbar'
import { paginationMeta }   from '@/utils/paginationMeta'
import CategoryDrawer       from '@/components/drawers/CategoryDrawer.vue'

interface Emit {
  (e: 'refresh'): void
}
const emit = defineEmits<Emit>()

const authStore = useAuthStore()
const store     = useCategoryStore()
const { showSnackbar } = useSnackbar()

// ── Permission entity prefix (matches PermissionSeeder key) ───────────────
const entity = ref('categories')

const hasPermission = (permission: string) =>
  computed(() => authStore.has_permission(permission))

const entityCreate = hasPermission(entity.value + '-create')
const entityEdit   = hasPermission(entity.value + '-edit')
const entityDelete = hasPermission(entity.value + '-delete')
const entityStatus = hasPermission(entity.value + '-status')

// ── Table state ───────────────────────────────────────────────────────────
const itemsPerPage  = ref(10)
const page          = ref(1)
const sortBy        = ref<string>()
const orderBy       = ref<string>()
const paramValue    = ref<any>({})
const searchQuery   = ref('')
const selectedRows  = ref<number[]>([])
const targetData    = ref<any>({})
const dataId        = ref(0)
const isDrawerVisible = ref(false)

// ── Fetch data ────────────────────────────────────────────────────────────
const getData = async () => {
  paramValue.value.searchQuery  = searchQuery.value
  paramValue.value.itemsPerPage = itemsPerPage.value
  paramValue.value.page         = page.value
  paramValue.value.sortBy       = sortBy.value  ?? 'created_at'
  paramValue.value.orderBy      = orderBy.value ?? 'desc'

  const data = await store.getCategoryData(paramValue.value)
  targetData.value = data
}

watch([page, itemsPerPage, sortBy, orderBy], async () => { await getData() })
watch([searchQuery], async () => { page.value = 1; await getData() })
onMounted(async () => { await getData() })

// ── Computed ──────────────────────────────────────────────────────────────
const items      = computed(() => targetData.value?.data?.rows || targetData.value?.rows || [])
const totalItems = computed(() => targetData.value?.data?.totalRows || targetData.value?.totalRows || 0)

const updateOptions = (options: any) => {
  page.value    = options.page
  sortBy.value  = options.sortBy[0]?.key
  orderBy.value = options.sortBy[0]?.order
}

// ── Deletion protection ───────────────────────────────────────────────────
const deletableRows = computed(() =>
  selectedRows.value.filter(id => {
    const item = items.value.find((i: any) => i.id === id)
    return item && !item.has_related_data
  })
)

const protectedRows = computed(() =>
  selectedRows.value.filter(id => {
    const item = items.value.find((i: any) => i.id === id)
    return item && item.has_related_data
  })
)

// ── Status colour helper ──────────────────────────────────────────────────
const statusColor = (status: string) => {
  if (status === '1') return 'success'   // active
  if (status === '2') return 'error'     // disabled/inactive
  if (status === '0') return 'warning'   // archive
  return 'secondary'
}

const statusLabel = (status: string) => {
  if (status === '1') return 'Active'
  if (status === '2') return 'Disabled'
  if (status === '0') return 'Archive'
  return 'Unknown'
}

// ── Refresh ───────────────────────────────────────────────────────────────
const handleRefresh = () => {
  getData()
  isDrawerVisible.value = false
  dataId.value = 0
  emit('refresh')
}

const handleSnackbar = (value: any) => {
  showSnackbar('', value.message)
  dataId.value = 0
}

// ── Delete with Swal ──────────────────────────────────────────────────────
const handleDelete = async (value: number | number[]) => {
  const ids     = Array.isArray(value) ? value : [value]
  const payload = ids.length > 1 ? { ids } : { id: ids[0] }

  const result = await Swal.fire({
    title: 'Delete Category?',
    text: 'Are you sure you want to delete? This cannot be undone.',
    icon: 'warning',
    showCancelButton: true,
    confirmButtonColor: '#3085d6',
    cancelButtonColor: '#d33',
    confirmButtonText: 'Yes, Delete it!',
  })

  if (result.isConfirmed) {
    const response = await store.categoryDelete(payload)
    showSnackbar('', response.success ? response.message : response.errors)
    selectedRows.value = []
    getData()
    emit('refresh')
  }
}

// ── Status change with Swal ───────────────────────────────────────────────
const handleStatusChange = async (item: any, newStatus: '0' | '1' | '2') => {
  const labels: Record<string, string> = {
    '0': 'Archive',
    '1': 'Activate',
    '2': 'Disable',
  }

  const result = await Swal.fire({
    title: `${labels[newStatus]} Category?`,
    text: `Are you sure you want to ${labels[newStatus].toLowerCase()} "${item.name}"?`,
    icon: 'warning',
    showCancelButton: true,
    confirmButtonColor: '#3085d6',
    cancelButtonColor: '#d33',
    confirmButtonText: `Yes, ${labels[newStatus]}!`,
  })

  if (result.isConfirmed) {
    const payload  = { ...item, newStatus }
    const response = await store.categoryStatusChange(payload)
    showSnackbar('', response.success ? response.message : response.errors)
    handleRefresh()
  }
}

// ── Edit ──────────────────────────────────────────────────────────────────
const editItem = async (id: number) => {
  dataId.value = 0
  await nextTick()
  dataId.value          = id
  isDrawerVisible.value = true
}

// ── Table headers ─────────────────────────────────────────────────────────
const headers = [
  { title: '',        key: 'data-table-select', width: '48px' },
  { title: '#',       key: 'id',             width: '60px' },
  { title: 'Name',    key: 'name'                           },
  { title: 'Slug',    key: 'slug'                           },
  { title: 'Keyword', key: 'api_keyword'                    },
  { title: 'Status',  key: 'status',         sortable: false },
  { title: 'Actions', key: 'actions',        sortable: false, align: 'end' },
]
</script>

<template>
  <VCard>
    <!-- ── Toolbar ─────────────────────────────────────────────── -->
    <VCardText>
      <div class="d-flex flex-wrap gap-4 align-center">

        <!-- Search -->
        <VTextField
          v-model="searchQuery"
          placeholder="Search categories..."
          prepend-inner-icon="ri-search-line"
          clearable
          hide-details
          density="compact"
          style="max-width: 300px"
        />

        <VSpacer />

        <!-- Protected rows chip -->
        <VChip
          v-if="protectedRows.length"
          color="warning"
          size="small"
          class="me-2"
        >
          <VIcon start icon="ri-lock-line" />
          {{ protectedRows.length }} protected
        </VChip>

        <!-- Bulk Delete -->
        <VBtn
          v-if="deletableRows.length && entityDelete.value"
          color="error"
          variant="outlined"
          size="small"
          prepend-icon="ri-delete-bin-line"
          @click="handleDelete(deletableRows)"
        >
          Delete ({{ deletableRows.length }})
        </VBtn>

        <!-- Add Button — permission gated -->
        <VBtn
          v-if="entityCreate.value"
          color="primary"
          prepend-icon="ri-add-line"
          @click="() => { dataId = 0; isDrawerVisible = true }"
        >
          Add Category
        </VBtn>

      </div>
    </VCardText>

    <!-- ── Table ────────────────────────────────────────────────── -->
    <VDataTableServer
      v-model="selectedRows"
      :headers="headers"
      :items="items"
      :items-length="totalItems"
      :loading="store.isLoading"
      :page="page"
      :items-per-page="itemsPerPage"
      item-value="id"
      show-select
      hover
      @update:options="updateOptions"
    >

      <!-- ID -->
      <template #item.id="{ item }">
        <span class="text-medium-emphasis text-body-2">{{ item.id }}</span>
      </template>

      <!-- Name -->
      <template #item.name="{ item }">
        <div class="d-flex align-center gap-2">
          <VAvatar size="30" color="primary" variant="tonal" rounded>
            <VIcon size="14" :icon="item.icon || 'ri-price-tag-3-line'" />
          </VAvatar>
          <span class="font-weight-medium">{{ item.name }}</span>
        </div>
      </template>

      <!-- Slug -->
      <template #item.slug="{ item }">
        <VChip size="small" label>{{ item.slug }}</VChip>
      </template>

      <!-- Status chip + click to change -->
      <template #item.status="{ item }">
        <VMenu v-if="entityStatus.value">
          <template #activator="{ props }">
            <VChip
              v-bind="props"
              :color="statusColor(item.status)"
              size="small"
              label
              style="cursor: pointer"
            >
              {{ statusLabel(item.status) }}
              <VIcon end size="12" icon="ri-arrow-down-s-line" />
            </VChip>
          </template>
          <VList density="compact" nav>
            <VListItem
              v-if="item.status !== '1'"
              prepend-icon="ri-checkbox-circle-line"
              title="Activate"
              @click="handleStatusChange(item, '1')"
            />
            <VListItem
              v-if="item.status !== '2'"
              prepend-icon="ri-close-circle-line"
              title="Disable"
              @click="handleStatusChange(item, '2')"
            />
            <VListItem
              v-if="item.status !== '0'"
              prepend-icon="ri-archive-line"
              title="Archive"
              @click="handleStatusChange(item, '0')"
            />
          </VList>
        </VMenu>
        <VChip v-else :color="statusColor(item.status)" size="small" label>
          {{ statusLabel(item.status) }}
        </VChip>
      </template>

      <!-- Actions -->
      <template #item.actions="{ item }">
        <div class="d-flex justify-end gap-1">

          <!-- Edit -->
          <VTooltip text="Edit" location="top">
            <template #activator="{ props }">
              <VBtn
                v-if="entityEdit.value"
                v-bind="props"
                icon size="small" variant="text"
                color="primary"
                @click="editItem(item.id)"
              >
                <VIcon size="18" icon="ri-edit-line" />
              </VBtn>
            </template>
          </VTooltip>

          <!-- Delete — show lock icon if has_related_data -->
          <VTooltip
            :text="item.has_related_data ? 'Cannot delete — has related data' : 'Delete'"
            location="top"
          >
            <template #activator="{ props }">
              <VBtn
                v-if="entityDelete.value"
                v-bind="props"
                icon size="small" variant="text"
                :color="item.has_related_data ? 'default' : 'error'"
                :disabled="item.has_related_data"
                @click="!item.has_related_data && handleDelete(item.id)"
              >
                <VIcon
                  size="18"
                  :icon="item.has_related_data ? 'ri-lock-line' : 'ri-delete-bin-line'"
                />
              </VBtn>
            </template>
          </VTooltip>

        </div>
      </template>

      <!-- Empty state -->
      <template #no-data>
        <div class="text-center pa-8">
          <VIcon size="52" color="disabled" icon="ri-price-tag-3-line" />
          <p class="text-body-1 font-weight-medium mt-3">No categories found</p>
          <p class="text-body-2 text-medium-emphasis mb-4">
            Try adjusting your search or add a new category
          </p>
          <VBtn
            v-if="entityCreate.value"
            color="primary"
            variant="tonal"
            prepend-icon="ri-add-line"
            @click="() => { dataId = 0; isDrawerVisible = true }"
          >
            Add first category
          </VBtn>
        </div>
      </template>

      <!-- Pagination footer -->
      <template #bottom>
        <VDivider />
        <div class="d-flex align-center justify-space-between flex-wrap gap-3 pa-4">

          <!-- Rows per page -->
          <div class="d-flex align-center gap-2">
            <span class="text-body-2 text-medium-emphasis">Rows per page:</span>
            <VSelect
              v-model="itemsPerPage"
              :items="[10, 20, 50, { title: 'All', value: -1 }]"
              density="compact"
              variant="outlined"
              hide-details
              style="width: 90px"
            />
          </div>

          <!-- Showing X–Y of Z -->
          <span class="text-body-2 text-medium-emphasis">
            {{ paginationMeta({ page, itemsPerPage }, totalItems) }}
          </span>

          <!-- Prev / Next -->
          <div class="d-flex gap-1">
            <VBtn
              icon size="small" variant="text"
              :disabled="page <= 1"
              @click="page--"
            >
              <VIcon icon="ri-arrow-left-s-line" />
            </VBtn>
            <VBtn
              icon size="small" variant="text"
              :disabled="page * itemsPerPage >= totalItems"
              @click="page++"
            >
              <VIcon icon="ri-arrow-right-s-line" />
            </VBtn>

// resources/ts/composables/useCategoryExport.ts
// Generated by: generate-export category
// Pattern from: composables/useSwipeRequestExport.ts

import dayjs from 'dayjs'
import * as ExcelJS from 'exceljs'

const fmtDate  = (d: string | null) => d ? dayjs(d).format('DD-MM-YYYY') : '—'
const statusLabel = (s: string) =>
  s === '1' ? 'Active' : s === '0' ? 'Archive' : 'Disabled'

// Status cell colours (ARGB)
const statusColors: Record<string, string> = {
  '1': 'FF2e7d32',  // green — active
  '2': 'FFc62828',  // red   — disabled
  '0': 'FFe65100',  // amber — archive
}

export function useCategoryExport() {

  const toRow = (item: any) => ({
    id:          item.id,
    name:        item.name,
    slug:        item.slug,
    api_keyword: item.api_keyword,
    status:      statusLabel(item.status),
    articles:    item.articles_count ?? 0,
    created_at:  fmtDate(item.created_at),
  })

  const exportExcel = async (data: any[]) => {
    const wb = new ExcelJS.Workbook()
    const ws = wb.addWorksheet('Categories')

    ws.columns = [
      { header: 'ID',          key: 'id',          width: 8  },
      { header: 'Name',        key: 'name',         width: 25 },
      { header: 'Slug',        key: 'slug',         width: 25 },
      { header: 'API Keyword', key: 'api_keyword',  width: 22 },
      { header: 'Status',      key: 'status',       width: 12 },
      { header: 'Articles',    key: 'articles',     width: 12 },
      { header: 'Created',     key: 'created_at',   width: 14 },
    ]

    // Header styling
    ws.getRow(1).eachCell(cell => {
      cell.font = { bold: true, color: { argb: 'FFFFFFFF' } }
      cell.fill = { type: 'pattern', pattern: 'solid', fgColor: { argb: 'FF6200ea' } }
      cell.border = {
        bottom: { style: 'thin', color: { argb: 'FFFFFFFF' } },
      }
    })

    // Freeze header + auto-filter
    ws.views = [{ state: 'frozen', ySplit: 1 }]
    ws.autoFilter = { from: 'A1', to: 'G1' }

    // Data rows
    data.forEach((item, idx) => {
      const row = ws.addRow(toRow(item))
      // Colour-code status cell
      const statusCell = row.getCell('status')
      const bg = statusColors[item.status] || 'FF9E9E9E'
      statusCell.fill = { type: 'pattern', pattern: 'solid', fgColor: { argb: bg } }
      statusCell.font = { color: { argb: 'FFFFFFFF' }, bold: true }
      // Alternating row bg
      if (idx % 2 === 0) {
        row.eachCell({ includeEmpty: true }, cell => {
          if (!cell.fill || (cell.fill as any).fgColor?.argb === undefined) {
            cell.fill = { type: 'pattern', pattern: 'solid', fgColor: { argb: 'FFF5F5F5' } }
          }
        })
      }
    })

    const buffer = await wb.xlsx.writeBuffer()
    const blob   = new Blob([buffer], { type: 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet' })
    const url    = URL.createObjectURL(blob)
    const a      = document.createElement('a')
    a.href       = url
    a.download   = `categories-${dayjs().format('YYYY-MM-DD')}.xlsx`
    a.click()
    URL.revokeObjectURL(url)
  }

  const exportCsv = (data: any[]) => {
    const headers = ['ID', 'Name', 'Slug', 'API Keyword', 'Status', 'Articles', 'Created']
    const rows    = data.map(item => {
      const r = toRow(item)
      return [r.id, r.name, r.slug, r.api_keyword, r.status, r.articles, r.created_at]
        .map(v => `"${String(v).replace(/"/g, '""')}"`)
        .join(',')
    })
    const csv  = [headers.join(','), ...rows].join('\n')
    const blob = new Blob([csv], { type: 'text/csv' })
    const url  = URL.createObjectURL(blob)
    const a    = document.createElement('a')
    a.href     = url
    a.download = `categories-${dayjs().format('YYYY-MM-DD')}.csv`
    a.click()
    URL.revokeObjectURL(url)
  }

  return { exportExcel, exportCsv }
}

            ├── Calls store:        categoryDelete (Swal confirmed)
            ├── Calls store:        categoryStatusChange (Swal confirmed)
            │
            └── CategoryDrawer.vue
                    │
                    │  Props:    isDrawerOpen, dataId (0=create, >0=edit)
                    │  Emits:    update:isDrawerOpen, refresh, update:snackbarMessage
                    │
                    ├── Watch dataId  →  getCategoryDetails  →  fill form
                    ├── Watch isOpen  →  fetch dropdown data
                    ├── Auto-slug     →  from name in create mode
                    ├── serverErrors  →  per-field from 422 response
                    └── onSubmit      →  manageCategory  →  emit refresh


STORE (category.ts)
    │
    └── apiCall wrapper  →  ApiService.setHeader()  →  axios
            │
            └── All 7 methods use same pattern:
                  GET  →  URLSearchParams query string
                  POST →  pass data object directly
```

### State Flow Rule

```
User interaction  →  Swal confirm (if destructive)
                  →  Store action called
                  →  apiCall wrapper  →  ApiService  →  Laravel /api/*
                  →  Response updates UI or shows serverErrors
                  →  emit('refresh') triggers parent to re-fetch stats
```
